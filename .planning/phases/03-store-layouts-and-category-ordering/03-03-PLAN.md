From 03-RESEARCH.md — svelte-dnd-action usage (Svelte 5 syntax, CRITICAL):
```typescript
import { dndzone, type DndEvent } from 'svelte-dnd-action'

let categories = $state<Array<{id: string; name: string; position: number}>>([])

function handleConsider(e: CustomEvent<DndEvent<typeof categories[0]>>) {
  categories = e.detail.items
}
function handleFinalize(e: CustomEvent<DndEvent<typeof categories[0]>>) {
  categories = e.detail.items
  persistOrder(categories)  // batch upsert to store_layouts
}
```
```svelte
<!-- Use onconsider/onfinalize (Svelte 5 prop syntax, NOT on:consider/on:finalize) -->
<div
  use:dndzone={{ items: categories, flipDurationMs: 200, dragStartThreshold: 1, delayTouchStart: true }}
  onconsider={handleConsider}
  onfinalize={handleFinalize}
>
  {#each categories as category (category.id)}
    <DraggableCategoryRow {category} />
  {/each}
</div>
```

From 03-RESEARCH.md — gap-of-10 position persistence:
```typescript
async function persistOrder(ordered: Array<{id: string}>) {
  const updates = ordered.map((cat, i) => ({ id: cat.id, position: (i + 1) * 10 }))
  await supabase.from('store_layouts').upsert(updates, { onConflict: 'store_id,category_id' })
}
// For default categories: upsert to categories table (id + position)
```

**src/lib/queries/stores.ts — new file:**

Export the following mutations (following the Phase 2 mutation pattern from items.ts):

- `createStoreMutation(supabase, householdId)` — inserts into stores table, invalidates ['stores', householdId]
- `deleteStoreMutation(supabase, householdId)` — deletes from stores by id, invalidates ['stores', householdId]
- `createCategoryMutation(supabase, householdId)` — inserts into categories, invalidates categoriesQueryKey. ALSO after insert: fetch all stores for household, batch-insert store_layouts rows for the new category (position = max existing + 10 for each store). This prevents Pitfall 5.
- `updateCategoryMutation(supabase, householdId)` — updates name on categories by id, optimistic update on categoriesQueryKey
- `deleteCategoryMutation(supabase, householdId)` — deletes category by id (cascade handles store_layouts and sets list_items.category_id to null), invalidates categoriesQueryKey
- `reorderDefaultCategoriesMutation(supabase, householdId)` — upsert to categories table: `supabase.from('categories').upsert(updates)` where updates = [{id, position: (i+1)*10}]. Invalidates categoriesQueryKey.
- `reorderStoreCategoriesMutation(supabase, storeId, householdId)` — upsert to store_layouts: `supabase.from('store_layouts').upsert(updates, { onConflict: 'store_id,category_id' })` where updates = [{store_id: storeId, category_id: cat.id, position: (i+1)*10}].
