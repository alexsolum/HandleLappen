```typescript
const LONG_PRESS_MS = 500
const MOVE_CANCEL_PX = 8  // match swipe.ts threshold

function startLongPress(e: PointerEvent) {
  const startX = e.clientX, startY = e.clientY
  longPressTimer = setTimeout(() => onLongPress?.(), LONG_PRESS_MS)
  function cancelOnMove(me: PointerEvent) {
    if (Math.abs(me.clientX - startX) > MOVE_CANCEL_PX ||
        Math.abs(me.clientY - startY) > MOVE_CANCEL_PX) cancelLongPress()
  }
  window.addEventListener('pointermove', cancelOnMove, { once: false })
  window.addEventListener('pointerup', () => {
    cancelLongPress()
    window.removeEventListener('pointermove', cancelOnMove)
  }, { once: true })
}
```

CRITICAL: The onpointerdown handler for long-press must NOT call e.preventDefault() before the tap threshold — doing so would break the click event and check-off. Start the timer in pointerdown; let the normal click proceed if released before 500ms.

From 03-RESEARCH.md — Optimistic assign category mutation:
```typescript
export function createAssignCategoryMutation(supabase, listId) {
  const queryClient = useQueryClient()
  const queryKey = itemsQueryKey(listId)
  return createMutation(() => ({
    mutationFn: async ({ itemId, categoryId }) => {
      const { error } = await supabase.from('list_items').update({ category_id: categoryId }).eq('id', itemId)
      if (error) throw error
    },
    onMutate: async ({ itemId, categoryId }) => {
      await queryClient.cancelQueries({ queryKey })
      const previous = queryClient.getQueryData(queryKey)
      queryClient.setQueryData(queryKey, (old = []) =>
        old.map(item => item.id === itemId ? { ...item, category_id: categoryId } : item)
      )
      return { previous }
    },
    onError: (_err, _vars, context) => {
      if (context?.previous) queryClient.setQueryData(queryKey, context.previous)
    },
    onSettled: () => queryClient.invalidateQueries({ queryKey }),
  }))
}
```
The optimistic update immediately changes category_id → the $derived groupedItems recomputes → item moves to new group visually without waiting for server response.

```typescript
let longPressTimer: ReturnType<typeof setTimeout> | null = null
const LONG_PRESS_MS = 500
const MOVE_CANCEL_PX = 8

function startLongPress(e: PointerEvent) {
  const startX = e.clientX
  const startY = e.clientY
  let cleanupMove: (() => void) | null = null

  function cancelOnMove(me: PointerEvent) {
    const dx = Math.abs(me.clientX - startX)
    const dy = Math.abs(me.clientY - startY)
    if (dx > MOVE_CANCEL_PX || dy > MOVE_CANCEL_PX) {
      cancelLongPress()
      if (cleanupMove) { cleanupMove(); cleanupMove = null }
    }
  }

  longPressTimer = setTimeout(() => {
    onLongPress?.()
    if (cleanupMove) { cleanupMove(); cleanupMove = null }
  }, LONG_PRESS_MS)

  window.addEventListener('pointermove', cancelOnMove)
  cleanupMove = () => window.removeEventListener('pointermove', cancelOnMove)

  window.addEventListener('pointerup', () => {
    cancelLongPress()
    if (cleanupMove) { cleanupMove(); cleanupMove = null }
  }, { once: true })
}

function cancelLongPress() {
  if (longPressTimer) { clearTimeout(longPressTimer); longPressTimer = null }
}
```
