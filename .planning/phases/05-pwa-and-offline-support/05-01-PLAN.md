**src/service-worker.ts** — Create new file:
```typescript
/// <reference lib="WebWorker" />
import { cleanupOutdatedCaches, precacheAndRoute } from 'workbox-precaching'
import { NavigationRoute, registerRoute } from 'workbox-routing'
import { CacheFirst, NetworkFirst } from 'workbox-strategies'
import { ExpirationPlugin } from 'workbox-expiration'
import { CacheableResponsePlugin } from 'workbox-cacheable-response'

declare let self: ServiceWorkerGlobalScope

// Injected by vite-pwa at build time — do not remove
precacheAndRoute(self.__WB_MANIFEST)
cleanupOutdatedCaches()

// App shell navigation: CacheFirst (assets already precached)
registerRoute(
  new NavigationRoute(new CacheFirst({ cacheName: 'app-shell' }))
)

// Supabase REST reads: NetworkFirst with 5s timeout
// Excludes /auth/ endpoints to avoid caching tokens (Pitfall 3 in RESEARCH.md)
registerRoute(
  ({ url }: { url: URL }) =>
    url.hostname.endsWith('.supabase.co') &&
    url.pathname.startsWith('/rest/v1/') &&
    !url.pathname.startsWith('/auth/'),
  new NetworkFirst({
    cacheName: 'supabase-rest',
    networkTimeoutSeconds: 5,
    plugins: [
      new ExpirationPlugin({ maxEntries: 200, maxAgeSeconds: 60 * 60 * 24 * 7 }),
      new CacheableResponsePlugin({ statuses: [0, 200] })
    ]
  }),
  'GET'
)

// Activate immediately on update
self.addEventListener('message', (event) => {
  if (event.data?.type === 'SKIP_WAITING') self.skipWaiting()
})
```

Note: BackgroundSync route for check-off PATCH is intentionally NOT in this service worker. Per RESEARCH.md open question 3, only the custom idb-keyval queue (plan 05-02) handles check-off offline — mixing Workbox BackgroundSync for PATCH would cause double-drain on Safari. The SW handles only reads (GET) and app shell.
